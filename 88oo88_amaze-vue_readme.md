### 使用

```bash
    npm install amaze-vue --save
```

```js
    import Vue from 'vue';
    import AmazeVue from 'amaze-vue';
    import 'amaze-vue/dist/amaze-vue.css';

    Vue.use(AmazeVue);
```

### 开发

```bash
    npm i
    npm run dev
```
